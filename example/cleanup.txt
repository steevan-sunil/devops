#!/bin/bash

# Specify the directory to clean up
DIRECTORY="d/devops/"

# Check if the directory exists
if [ -d "$DIRECTORY" ]; then
  # Find and delete files older than 3 days
  find "$DIRECTORY" -type f -mtime +3 -exec rm -f {} \;
  
  # Print completion message
  echo "Files older than 3 days in $DIRECTORY have been deleted."
else
  echo "Directory $DIRECTORY does not exist."
fi
